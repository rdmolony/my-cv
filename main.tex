% Command for skill pills
\definecolor{pillcolor}{RGB}{229,231,235}
\newcommand{\skillpill}[1]{%
  \tikz[baseline=(text.base)]\node[
    rectangle,
    rounded corners=15pt,
    fill=pillcolor,
    text=textcolor,
    font=\small,
    inner sep=8pt,
    inner ysep=4pt
  ] (text) {#1};%
}

\newcommand{\skills}[1]{%
  \foreach \skill in {#1} {%
    \skillpill{\skill}\hspace{0pt}%
  }%
}
